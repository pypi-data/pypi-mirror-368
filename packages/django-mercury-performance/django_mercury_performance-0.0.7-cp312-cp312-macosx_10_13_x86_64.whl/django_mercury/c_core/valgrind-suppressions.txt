# Valgrind suppression file for Django Mercury C Core
# These are known false positives that can be safely ignored

# libunwind writes uninitialized bytes during stack unwinding
# This is a known issue in libunwind and is harmless
{
   libunwind_uninitialized_write
   Memcheck:Param
   write(buf)
   fun:syscall
   ...
   fun:_ULx86_64_step
   fun:capture_stack_trace*
   fun:start_performance_monitoring_enhanced
}

# libunwind invalid file descriptor warnings
# These occur when libunwind looks for debug symbols
{
   libunwind_invalid_fd
   Memcheck:Param
   close(fd)
   fun:close
   obj:*/libunwind.so.*
}
